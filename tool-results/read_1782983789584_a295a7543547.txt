    68тЖТ    68тЖТ// Progressive difficulty: freeze window shrinks and music gets shorter each round.
    69тЖТ    69тЖТ// Round 1 = 1800ms window, shrinks 120ms per round, clamped to 500ms minimum.
    70тЖТ    70тЖТconst BASE_WINDOW = 1800
    71тЖТ    71тЖТconst MIN_WINDOW = 500
    72тЖТ    72тЖТconst WINDOW_SHRINK = 120
    73тЖТ    73тЖТ
    74тЖТ    74тЖТfunction getFreezeWindow(round: number) {
    75тЖТ    75тЖТ  return Math.max(MIN_WINDOW, BASE_WINDOW - (round - 1) * WINDOW_SHRINK)
    76тЖТ    76тЖТ}
    77тЖТ    77тЖТ
    78тЖТ    78тЖТ// Music duration also shortens with rounds: starts 4000-7500ms, shrinks toward 2500-4500ms.
    79тЖТ    79тЖТfunction randMusicDuration(round: number) {
    80тЖТ    80тЖТ  const shrink = Math.min(1500, (round - 1) * 120)
    81тЖТ    81тЖТ  const musicMin = Math.max(2500, 4000 - shrink)
    82тЖТ    82тЖТ  const musicMax = Math.max(4500, 7500 - shrink)
    83тЖТ    83тЖТ  return musicMin + Math.random() * (musicMax - musicMin)
    84тЖТ    84тЖТ}

    88тЖТ    88тЖТ  const d = new Date()

    90тЖТ    90тЖТ}

    92тЖТ    92тЖТfunction mulberry32(seed: number) {
    93тЖТ    93тЖТ  let a = seed
    94тЖТ    94тЖТ  return function () {
    95тЖТ    95тЖТ    a |= 0
    96тЖТ    96тЖТ    a = (a + 0x6d2b79f5) | 0
    97тЖТ    97тЖТ    let t = a
    98тЖТ    98тЖТ    t = Math.imul(t ^ (t >>> 15), t | 1)
    99тЖТ    99тЖТ    t ^= t + Math.imul(t ^ (t >>> 7), t | 61)
   100тЖТ   100тЖТ    return ((t ^ (t >>> 14)) >>> 0) / 4294967296
   101тЖТ   101тЖТ  }
   102тЖТ   102тЖТ}

   104тЖТ   104тЖТfunction dailyMusicDuration(round: number): number {
   105тЖТ   105тЖТ  const shrink = Math.min(1500, (round - 1) * 120)
   106тЖТ   106тЖТ  const musicMin = Math.max(2500, 4000 - shrink)
   107тЖТ   107тЖТ  const musicMax = Math.max(4500, 7500 - shrink)
   108тЖТ   108тЖТ  return musicMin + dailyRand() * (musicMax - musicMin)
   109тЖТ   109тЖТ}

   111тЖТ   111тЖТlet _dailyRand: (() => number) | null = null
   112тЖТ   112тЖТfunction dailyRand(): number {
   113тЖТ   113тЖТ  if (!_dailyRand) _dailyRand = mulberry32(getTodaySeed())
   114тЖТ   114тЖТ  return _dailyRand()
   115тЖТ   115тЖТ}
